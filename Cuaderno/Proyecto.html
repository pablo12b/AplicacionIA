import os

import yaml

# Ruta donde están las imágenes y etiquetas juntas
src_dir = Path("dataset-main/images")

# Crear estructura para YOLO
yolo_base = Path("YOLO_ready_dataset")
(yolo_base / "images/train").mkdir(parents=True, exist_ok=True)
(yolo_base / "images/val").mkdir(parents=True, exist_ok=True)
(yolo_base / "labels/train").mkdir(parents=True, exist_ok=True)
(yolo_base / "labels/val").mkdir(parents=True, exist_ok=True)

# Listar todas las imágenes (.jpg)
all_images = list(src_dir.glob("*.jpg"))

# Dividir train/val
train_imgs, val_imgs = train_test_split(all_images, test_size=0.2, random_state=42)

def copy_files(img_list, split):
    for img_path in img_list:
        label_path = img_path.with_suffix(".txt")
        # Copiar imagen
        shutil.copy(img_path, yolo_base / f"images/{split}" / img_path.name)
        # Copiar etiqueta si existe
        if label_path.exists():
            shutil.copy(label_path, yolo_base / f"labels/{split}" / label_path.name)
        else:
            print(f"Etiqueta no encontrada para {img_path.name}")

copy_files(train_imgs, "train")
copy_files(val_imgs, "val")

print(f"Dataset organizado: {len(train_imgs)} imágenes en train, {len(val_imgs)} en val")

    "path": str(yolo_base),

import os

def get_classes_from_labels(label_dir):
    classes_found = set()
    for filename in os.listdir(label_dir):
        if filename.endswith(".txt"):
            filepath = os.path.join(label_dir, filename)
            with open(filepath, 'r') as f:
                for line in f:
                    if line.strip():
                        class_id = int(line.strip().split()[0])

def construir_df(split):
    label_path = f"YOLO_ready_dataset/labels/{split}"
    image_path = f"YOLO_ready_dataset/images/{split}"

    data = []
    for label_file in os.listdir(label_path):
        label_file_path = os.path.join(label_path, label_file)
        with open(label_file_path, 'r') as f:
            for line in f:
                if not line.strip():
                    continue
                parts = line.strip().split()
                class_id = int(float(parts[0]))
                if class_id >= len(class_names):
                    continue  # salta clases inválidas

                class_name = class_names[class_id]
                image_name = label_file.replace('.txt', '')
                image_file = os.path.join(image_path, f"{image_name}.jpg")
                if not os.path.exists(image_file):
                    image_file = os.path.join(image_path, f"{image_name}.png")
                if not os.path.exists(image_file):
                    continue  # imagen no encontrada

                data.append({
                    'filename': image_file,
                    'class': class_name
                })
    return pd.DataFrame(data)

def predict_image(img_path, model, class_indices):
    # Cargar y preprocesar la imagen
    img = image.load_img(img_path, target_size=(224, 224))
    img_array = image.img_to_array(img) / 255.0  # Normalizar igual que en entrenamiento
    img_array = np.expand_dims(img_array, axis=0)  # Añadir batch dimension

    # Predecir
    preds = model.predict(img_array)
    pred_class_index = np.argmax(preds, axis=1)[0]
    pred_confidence = preds[0][pred_class_index]

    # Obtener nombre de clase según índice
    class_names = {v:k for k,v in class_indices.items()}
    pred_class_name = class_names[pred_class_index]

    # Mostrar resultado
    plt.imshow(img)
    plt.axis('off')
    plt.title(f"Predicción: {pred_class_name} ({pred_confidence*100:.2f}%)")
    plt.show()

    return pred_class_name, pred_confidence
